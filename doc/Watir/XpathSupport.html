def element_by_xpath(xpath)
  e = wd.find_element(:xpath, xpath)
  Watir.element_class_for(e.tag_name).new(self, :element, e)
rescue WebDriver::Error::WebDriverError
  BaseElement.new(self, :xpath, xpath)
end
